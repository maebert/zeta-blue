
export const InfoBox = ({ title, description, children }) => {
    const [isOpen, setIsOpen] = useState(false);

    return <div role="button" className="callout border rounded-md mb-3 overflow-hidden border-sky-500/20 bg-sky-50/50 dark:border-sky-500/30 dark:bg-sky-500/10 text-sky-900 dark:text-sky-200 cursor-default">
        <button
            className="relative not-prose w-full py-4 px-5 hover:bg-sky-100 hover:dark:bg-sky-800 mb-0"
            aria-expanded={isOpen}
            onClick={() => setIsOpen(!isOpen)}
        >
            <div className="w-full flex flex-row items-center content-center space-x-2">
                <Icon icon="lightbulb" size={16} />
                <p className="m-0 text-left font-medium text-sky-900 flex-1 dark:text-sky-200">{title}</p>
                <Icon icon={isOpen ? "chevron-up" : "chevron-down"} size={16} />
            </div>
            {description && <p className="text-left mt-2 text-sm text-sky-700 dark:text-sky-400">{description}</p>}
        </button>
        {isOpen && <div className="mx-5 overflow-x-auto text-sm text-sky-900 dark:text-sky-200">
            {children}
        </div>}
    </div>
};
